CREATE TABLE LotteryTickets (
    ticketID INT IDENTITY(1,1) PRIMARY KEY,   -- Tự tăng, khóa chính
    userID INT NULL,                          -- Cho phép NULL
    ticketNumber INT NULL,                    -- Cho phép NULL
    isWin BIT NOT NULL DEFAULT 0,             -- Mặc định 0
    isCheck BIT NOT NULL DEFAULT 0,           -- Mặc định 0
    purchaseTime DATETIME NOT NULL DEFAULT GETDATE() -- Mặc định là thời điểm hiện tại
);
